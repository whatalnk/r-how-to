trinker/wakefield

カテゴリごとにそれっぽいデータを生成してくれるパッケージ

d <- data.frame(
  "番号" = wakefield::id(10),
  "車" = wakefield::car(10), 
  "速度" = wakefield::speed_kph(10)
)

保存

library(openxlsx)
write.xlsx(d, file="car-random.xlsx")
